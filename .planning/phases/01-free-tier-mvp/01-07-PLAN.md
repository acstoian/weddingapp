       ```typescript
       import { Resend } from 'resend'

       const resend = new Resend(process.env.RESEND_API_KEY)

       // Simple React email template inline (no separate file needed for Phase 1)
       function PublishSuccessEmailHtml(title: string, url: string): string {
         return `<div style="font-family:sans-serif;max-width:600px">
           <h2>Invitatia ta este live!</h2>
           <p><strong>${title}</strong> este acum disponibila la:</p>
           <p><a href="${url}">${url}</a></p>
           <p>Poti distribui acest link invitatilor tai.</p>
         </div>`
       }

       export async function sendPublishSuccessEmail(to: string, title: string, url: string) {
         const { error } = await resend.emails.send({
           from: `Invitatii <no-reply@${process.env.RESEND_FROM_DOMAIN}>`,
           to: [to],
           subject: `Invitatia ta "${title}" este live!`,
           html: PublishSuccessEmailHtml(title, url),
         })
         if (error) console.error('[email] Resend error:', error)
         // Don't throw — email failure should not block the success response
       }
       ```

  <behavior>
    - SSE stream emits `data: {"status":"BUILDING","url":null}\n\n` while Vercel readyState is QUEUED/BUILDING/INITIALIZING
    - SSE stream emits `data: {"status":"READY","url":"https://..."}\n\n` when Vercel readyState is READY; closes stream
    - SSE stream emits `data: {"status":"ERROR","url":null}\n\n` when Vercel readyState is ERROR; closes stream
    - On READY: DB updated with `status='LIVE'`, `liveUrl=url`, `lastPublishedAt=now()`
    - On READY: `sendPublishSuccessEmail` called with user's Clerk email
    - On ERROR: DB updated with `status='FAILED'`
    - SSE response headers: `Content-Type: text/event-stream`, `Cache-Control: no-cache`
    - `maxDuration = 120` prevents serverless timeout for normal deployments
  </behavior>
  <action>
    1. `src/app/api/deploy-status/[deploymentId]/route.ts`:
       - `export const dynamic = 'force-dynamic'`
       - `export const runtime = 'nodejs'`
       - `export const maxDuration = 120`
       - Authenticate; parse `deploymentId` from params
       - Look up the invitation in DB by `vercelDeployId = deploymentId` to get invitationId, userId, title, liveUrl
       - Get user's email via Clerk: `const user = await clerkClient().users.getUser(userId)`
       - Create `ReadableStream` per RESEARCH.md Pattern 6 exactly:
         ```typescript
         const stream = new ReadableStream({
           async start(controller) {
             const encoder = new TextEncoder()
             const send = (data: object) =>
               controller.enqueue(encoder.encode(`data: ${JSON.stringify(data)}\n\n`))

             let done = false
             while (!done) {
               const status = await getDeploymentService().pollStatus(deploymentId)

               if (status === 'READY') {
                 // Get URL from Vercel
                 const urlRes = await fetch(`https://api.vercel.com/v13/deployments/${deploymentId}`, {
                   headers: { Authorization: `Bearer ${process.env.VERCEL_TOKEN}` }
                 })
                 const { url } = await urlRes.json()
                 const liveUrl = `https://${url}`

                 // Update DB
                 await db.update(invitations)
                   .set({ status: 'LIVE', liveUrl, lastPublishedAt: new Date(), updatedAt: new Date() })
                   .where(eq(invitations.vercelDeployId, deploymentId))

                 // Send email (best-effort)
                 await sendPublishSuccessEmail(userEmail, invitationTitle, liveUrl)

                 send({ status: 'READY', url: liveUrl })
                 done = true
                 controller.close()

               } else if (status === 'ERROR') {
                 await db.update(invitations)
                   .set({ status: 'FAILED', updatedAt: new Date() })
                   .where(eq(invitations.vercelDeployId, deploymentId))

                 send({ status: 'ERROR', url: null })
                 done = true
                 controller.close()

               } else {
                 // QUEUED or BUILDING
                 send({ status: 'BUILDING', url: null })
                 await new Promise(r => setTimeout(r, 3000))
               }
             }
           }
         })

         return new Response(stream, {
           headers: {
             'Content-Type': 'text/event-stream',
             'Cache-Control': 'no-cache',
             Connection: 'keep-alive',
           }
         })
         ```
