```typescript
// src/components/editor/useAutosave.ts
export function useAutosave(invitationId: string, fields: InvitationFields, setStatus: (s: 'idle'|'saving'|'saved') => void) {
  const [debouncedFields] = useDebounce(fields, 2000)
  useEffect(() => {
    if (!debouncedFields) return
    setStatus('saving')
    fetch(`/api/invitations/${invitationId}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ fields: debouncedFields }),
    }).then(() => setStatus('saved'))
  }, [debouncedFields, invitationId])
}
```
